Here's a small useDebouncedSave hook. It batches changes and writes after the field goes quiet for 600ms, so every edit autosaves without a Save button:

typescript
// debounced autosave — fires 600ms after the last change
export function useDebouncedSave(value, save) {
  const t = useRef();
  useEffect(() => {
    if (t.current) clearTimeout(t.current);
    t.current = setTimeout(() => save(value), 600);
    return () => clearTimeout(t.current);
  }, [value]);
}
